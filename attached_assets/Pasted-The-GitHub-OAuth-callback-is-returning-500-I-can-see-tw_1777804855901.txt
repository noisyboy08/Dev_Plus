app.get('/api/auth/github/callback', async (req, res) => {
  try {
    console.log('=== CALLBACK HIT ===');
    console.log('Query params:', req.query);
    console.log('Session:', req.session);
    
    const { code, state } = req.query;
    
    if (!code) {
      console.log('ERROR: No code received');
      return res.status(400).send('No code received from GitHub');
    }
    
    console.log('Exchanging code for token...');
    const tokenResponse = await fetch(
      'https://github.com/login/oauth/access_token',
      {
        method: 'POST',
        headers: {
          'Accept': 'application/json',
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          client_id: process.env.GITHUB_CLIENT_ID,
          client_secret: process.env.GITHUB_CLIENT_SECRET,
          code: code,
          redirect_uri: process.env.GITHUB_CALLBACK_URL
        })
      }
    );
    
    const tokenData = await tokenResponse.json();
    console.log('Token response:', JSON.stringify(tokenData));
    
    if (tokenData.error) {
      console.log('GitHub error:', tokenData.error, tokenData.error_description);
      return res.status(500).send(`GitHub error: ${tokenData.error_description}`);
    }
    
    const accessToken = tokenData.access_token;
    
    const userResponse = await fetch('https://api.github.com/user', {
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'Accept': 'application/vnd.github.v3+json'
      }
    });
    
    const userData = await userResponse.json();
    console.log('User data:', userData.login, userData.id);
    
    req.session.userId = userData.id;
    req.session.user = {
      id: userData.id,
      login: userData.login,
      avatar_url: userData.avatar_url,
      access_token: accessToken
    };
    
    await new Promise((resolve, reject) => {
      req.session.save((err) => {
        if (err) reject(err);
        else resolve();
      });
    });
    
    console.log('Session saved, redirecting to dashboard');
    res.redirect('/dashboard');
    
  } catch (error) {
    console.error('=== CALLBACK ERROR ===');
    console.error(error);
    res.status(500).send(`Auth error: ${error.message}`);
  }
});
